Linux named service (DNS server)
__________________________________________________________________________________________________________________________________________________________________________________________________________________________________

/etc/named.conf                                              -> main configuration file for named
/var/named/somename.somedomain.zone                          -> configuration file for somename.somedomain

_________________________________________________________________________________________________________________________________________________________________________________________________________________________________________________________
named.conf__________________________________________________________________________________________________________________________________________________________________________________________________________________________________
// Default named.conf generated by install of bind-9.2.4-2
options {
	directory "/var/named";
	dump-file "/var/named/data/cache_dump.db";
	statistics-file "/var/named/data/named_stats.txt";
        forward first;
        forwarders {
                10.124.100.1;
        };
};
include "/etc/rndc.key";

zone "zimbrahost.wilddomain" in {
        type master;
        file "zimbrahost.wilddomain.zone";
        notify no;
};

                                        2008101011      ; serial

                                        1W              ; expiry
                                        1D )            ; minimum

                        1D IN NS        ns.somename.somedomain.
                        1D IN A         10.11.12.13
