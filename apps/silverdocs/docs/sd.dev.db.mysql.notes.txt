Creating triggers
----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

CREATE TRIGGER `Trigger_MyTable_Col1_Field1_Insert` BEFORE INSERT ON `MyTable`
  FOR EACH ROW BEGIN
    IF NEW.Field1 IS NULL THEN
      SET NEW.Field1 = 
        CASE NEW.Field2
          WHEN 'VAL1' THEN 0
          WHEN 'VAL2' THEN 1
          ELSE -1
        END;
    ELSE
      SET NEW.Field1 = 
        CASE NEW.Field3
          WHEN 0 THEN 'TEXT1'
          WHEN 1 THEN 'TEXT2' 
          ELSE 'UNDEFINED'
        END;
    END IF;
  END;
